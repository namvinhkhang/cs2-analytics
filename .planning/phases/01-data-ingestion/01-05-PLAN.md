       ```python
       class PandaScoreClient(BaseAPIClient):
           """PandaScore REST API client for CS2 (counterstrike) match results and player stats.

           Free tier rate limit: 1,000 req/hour = ~0.27 req/s.
           A 3.6-second sleep after EVERY request is mandatory to prevent 429 exhaustion.
           (Pitfall: asyncio.gather() without rate limiting fires all coroutines at once
           and exhausts the hourly budget in seconds — see RESEARCH.md Pitfall 1)

           Auth: Bearer token in Authorization header.
           CS2 game prefix: /csgo/ (PandaScore uses the legacy game slug)
           """

           BASE_URL: str = "https://api.pandascore.co"
           _semaphore: asyncio.Semaphore = asyncio.Semaphore(1)

           def _auth_headers(self) -> dict[str, str]:
               return {"Authorization": f"Bearer {self._api_key}"}

           async def get_recent_matches(
               self, page: int = 1, per_page: int = 50
           ) -> list[PandaScoreMatch]:
               """Fetch recent CS2 match results (past matches).

               Uses /csgo/matches/past endpoint; sleeps 3.6s after request.
               """
               data = await self.get(
                   "/csgo/matches/past",
                   page=page,
                   per_page=per_page,
               )
               await asyncio.sleep(3.6)  # MANDATORY: 1,000 req/hour free tier limit
               # PandaScore returns a list at the top level for list endpoints
               if isinstance(data, list):
                   records = data
               else:
                   records = data.get("results", [])
               return [PandaScoreMatch.model_validate(r) for r in records]

           async def get_players(
               self, page: int = 1, per_page: int = 50
           ) -> list[PandaScorePlayer]:
               """Fetch CS2 player profiles.

               Uses /csgo/players endpoint; sleeps 3.6s after request.
               """
               data = await self.get(
                   "/csgo/players",
                   page=page,
                   per_page=per_page,
               )
               await asyncio.sleep(3.6)  # MANDATORY: 1,000 req/hour free tier limit
               if isinstance(data, list):
                   records = data
               else:
                   records = data.get("results", [])
               return [PandaScorePlayer.model_validate(r) for r in records]

           async def ingest_matches(
               self,
               bucket: str,
               ingest_date: date,
               *,
               pages: int = 1,
               per_page: int = 50,
               region: str = "us-east-1",
           ) -> int:
               """Fetch match pages and write canonical Match records to S3.

               Returns count of matches written.
               """
               canonical_matches: list[Match] = []
               for page in range(1, pages + 1):
                   raw_matches = await self.get_recent_matches(page=page, per_page=per_page)
                   canonical_matches.extend([m.to_canonical() for m in raw_matches])

               y, m, d = ingest_date.year, ingest_date.month, ingest_date.day
               if canonical_matches:
                   write_parquet_to_s3(
                       records=models_to_records(canonical_matches),
                       schema=MATCH_SCHEMA,
                       bucket=bucket,
                       key=build_s3_key("pandascore", "matches", y, m, d),
                       region=region,
                   )
               logger.info("pandascore_matches_ingested", count=len(canonical_matches))
               return len(canonical_matches)

           async def ingest_players(
               self,
               bucket: str,
               ingest_date: date,
               *,
               pages: int = 1,
               per_page: int = 50,
               region: str = "us-east-1",
           ) -> int:
               """Fetch player pages and write canonical Player records to S3.

               Returns count of players written.
               Advanced stats (adr, kast) may be None on free tier — this is expected.
               """
               canonical_players: list[Player] = []
               recorded_at = ingest_date.isoformat()

               for page in range(1, pages + 1):
                   raw_players = await self.get_players(page=page, per_page=per_page)
                   for p in raw_players:
                       canonical_players.append(
                           p.to_canonical(
                               kills=None,   # per-match stats not available from /csgo/players
                               deaths=None,  # they require match-specific endpoints
                               adr=None,     # may be premium-only — treat as None
                               kd_ratio=None,
                               match_id=None,
                               recorded_at=recorded_at,
                           )
                       )

               y, m, d = ingest_date.year, ingest_date.month, ingest_date.day
               if canonical_players:
                   write_parquet_to_s3(
                       records=models_to_records(canonical_players),
                       schema=PLAYER_SCHEMA,
                       bucket=bucket,
                       key=build_s3_key("pandascore", "players", y, m, d),
                       region=region,
                   )
               logger.info("pandascore_players_ingested", count=len(canonical_players))
               return len(canonical_players)
       ```

    Critical notes:
    - asyncio.sleep(3.6) is INSIDE each get_*() method, NOT in ingest_*(). This ensures the
      sleep fires even when get_*() is called directly in tests — the sleep is part of the
      rate-limit contract, not an afterthought in the caller.
    - PandaScore list endpoints return a bare JSON array (not {"results": [...]}). The
      isinstance(data, list) check handles this — the base get() returns dict[str, Any]
      but for list endpoints the JSON root is an array which Python's json.loads() returns
      as a list. Note: this means the return type annotation on get() is approximate.
      Use a cast or ignore if mypy complains.
    - CS2 game prefix is "/csgo/" (legacy slug from when it was CS:GO) — do NOT use "/cs2/"
      as PandaScore has not migrated to the new slug for match endpoints as of research date.
    - All advanced stats (adr, kast) default to None in ingest_players() — these are
      per-match stats not available from the /csgo/players profile endpoint (Pitfall 3).
      A future plan can add per-match stat fetching against specific match IDs.
    - Do NOT use asyncio.gather() across multiple pages — sequential pages with sleep is safer
      and avoids burst exhaustion on the 1,000 req/hour free tier (Pitfall 1).
