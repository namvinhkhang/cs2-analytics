    ```python
    """Tests for BaseAPIClient retry and context manager behavior."""
    from __future__ import annotations
    import asyncio
    from typing import Any
    import httpx
    import pytest
    import respx
    from cs2_analytics.ingestion.base import BaseAPIClient


    class ConcreteClient(BaseAPIClient):
        """Minimal concrete subclass for testing BaseAPIClient in isolation."""
        BASE_URL: str = "https://test-api.example.com"
        _semaphore: asyncio.Semaphore = asyncio.Semaphore(10)  # high limit — don't block tests

        def _auth_headers(self) -> dict[str, str]:
            return {"X-Test": "test-key"}


    @pytest.mark.asyncio
    async def test_get_retries_on_429(respx_mock: respx.MockRouter) -> None:
        """get() retries after receiving HTTP 429."""
        call_count = 0

        def side_effect(request: httpx.Request) -> httpx.Response:
            nonlocal call_count
            call_count += 1
            if call_count < 2:
                return httpx.Response(429, json={"error": "rate limited"})
            return httpx.Response(200, json={"data": "ok"})

        respx_mock.get("https://test-api.example.com/test").mock(side_effect=side_effect)

        async with ConcreteClient(api_key="test") as client:
            result = await client.get("/test")

        assert result == {"data": "ok"}
        assert call_count == 2


    @pytest.mark.asyncio
    async def test_get_raises_after_max_attempts(respx_mock: respx.MockRouter) -> None:
        """get() raises HTTPStatusError after exhausting all retry attempts."""
        respx_mock.get("https://test-api.example.com/test").mock(
            return_value=httpx.Response(500, json={"error": "server error"})
        )

        async with ConcreteClient(api_key="test") as client:
            with pytest.raises(httpx.HTTPStatusError):
                await client.get("/test")


    @pytest.mark.asyncio
    async def test_context_manager_closes_client() -> None:
        """__aexit__ closes the underlying httpx.AsyncClient."""
        client = ConcreteClient(api_key="test")
        assert not client._client.is_closed
        async with client:
            pass
        assert client._client.is_closed
    ```
