Create airflow/Dockerfile:
```dockerfile
# Extends official Airflow 2.9 image to bake in cs2_analytics package and Slack provider.
# Using Dockerfile instead of _PIP_ADDITIONAL_REQUIREMENTS to avoid re-pip on every container start.
FROM apache/airflow:2.9-python3.12

USER root
# Copy package metadata and source for pip install
COPY pyproject.toml /opt/cs2-analytics/pyproject.toml
COPY src /opt/cs2-analytics/src

# Install local package (cs2_analytics) and Slack provider
# Pin apache-airflow-providers-slack==8.7.1 — version from constraints-2.9.3/constraints-3.12.txt
# (9.x requires Airflow >= 2.11 and will conflict)
RUN pip install --no-cache-dir /opt/cs2-analytics \
    "apache-airflow-providers-slack==8.7.1"

USER airflow
# Add src to PYTHONPATH so `from cs2_analytics.ingestion.faceit import FACEITClient` works
# in all containers (scheduler + worker + webserver share this anchor)
ENV PYTHONPATH="/opt/cs2-analytics/src:${PYTHONPATH}"
```

Full docker-compose.yml structure:
```yaml
---
# CS2 Analytics — Airflow CeleryExecutor stack
# Start: docker compose up airflow-init && docker compose up -d
# Requires: AIRFLOW_UID=$(id -u) in .env (Linux only, prevents volume permission issues)
version: "3.8"

x-airflow-common: &airflow-common
  build:
    context: ..
    dockerfile: airflow/Dockerfile
  image: cs2-airflow:2.9-python3.12
  env_file:
    - ../.env
  environment:
    AIRFLOW__CORE__EXECUTOR: CeleryExecutor
    AIRFLOW__DATABASE__SQL_ALCHEMY_CONN: postgresql+psycopg2://airflow:airflow@postgres/airflow
    AIRFLOW__CELERY__RESULT_BACKEND: db+postgresql://airflow:airflow@postgres/airflow
    AIRFLOW__CELERY__BROKER_URL: redis://:@redis:6379/0
    AIRFLOW__CORE__DAGS_FOLDER: /opt/airflow/dags
    AIRFLOW__CORE__FERNET_KEY: "${CS2_AIRFLOW_FERNET_KEY:-}"
    AIRFLOW__CORE__LOAD_EXAMPLES: "false"
    AIRFLOW__WEBSERVER__SECRET_KEY: "${CS2_AIRFLOW_SECRET_KEY:-dev_secret_change_me}"
    PYTHONPATH: /opt/cs2-analytics/src
  volumes:
    - ../airflow/dags:/opt/airflow/dags
    - ../logs:/opt/airflow/logs
    - ../src:/opt/cs2-analytics/src
  depends_on:
    postgres:
      condition: service_healthy
    redis:
      condition: service_healthy

services:
  postgres:
    image: postgres:16
    environment:
      POSTGRES_USER: airflow
      POSTGRES_PASSWORD: airflow
      POSTGRES_DB: airflow
    volumes:
      - postgres-db-volume:/var/lib/postgresql/data
    healthcheck:
      test: ["CMD", "pg_isready", "-U", "airflow"]
      interval: 10s
      retries: 5
      start_period: 5s
    restart: always

  redis:
    image: redis:7-alpine
    expose:
      - 6379
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 10s
      retries: 5
      start_period: 5s
    restart: always

  airflow-webserver:
    <<: *airflow-common
    command: webserver
    ports:
      - "8080:8080"
    healthcheck:
      test: ["CMD", "curl", "--fail", "http://localhost:8080/health"]
      interval: 30s
      timeout: 10s
      retries: 5
      start_period: 30s
    restart: always

  airflow-scheduler:
    <<: *airflow-common
    command: scheduler
    healthcheck:
      test: ["CMD", "airflow", "jobs", "check", "--job-type", "SchedulerJob", "--hostname", "$${HOSTNAME}"]
      interval: 30s
      timeout: 10s
      retries: 5
      start_period: 30s
    restart: always

  airflow-worker:
    <<: *airflow-common
    command: celery worker
    environment:
      DUMB_INIT_SETSID: "0"
    restart: always

  airflow-triggerer:
    <<: *airflow-common
    command: triggerer
    restart: always

  airflow-init:
    <<: *airflow-common
    command: >
      bash -c "
        airflow db migrate &&
        airflow users create
          --username admin
          --password admin
          --firstname Admin
          --lastname User
          --role Admin
          --email admin@example.com
      "
    restart: "no"

volumes:
  postgres-db-volume:
```
